Ext Server

The Ext Server test suite.

Ext.apply

空のオブジェクトに値が適用されること


        var src = {}, ret;

        ret = Ext.apply(src, {
            foo: 'bar'
        });

        src.should.equal(ret);
        src.foo.should.equal('bar');
デフォルト値が適用されること


        var src = {}, config, defaults, ret;

        config = {
            foo: 'bar'
        };

        defaults = {
            foo     : 'default value',
            hoge    : 'piyo'
        };

        ret = Ext.apply(src, config, defaults);

        src.should.equal(ret);
        src.foo.should.equal('bar');
        src.hoge.should.equal('piyo');

Ext.applyIf

未定義のキーに新たなキーを追加することができること


        var src, ret;

        src = {
            hoge: 'piyo'
        };

        ret = Ext.applyIf(src, {
            foo: 'bar'
        });

        src.should.equal(ret);
        src.foo.should.equal('bar');
        src.hoge.should.equal('piyo');
既に定義されているキーが上書きされないこと


        var src, config, ret;

        src = {
            foo: 'original'
        };

        config = {
            foo: 'bar'
        };

        ret = Ext.applyIf(src, config);

        src.should.equal(ret);
        src.foo.should.equal('original');

Ext.clone

オブジェクト型のクローンを生成し、値を変更してオリジナルに影響がないこと


        var o, c;

        o = {
            src: 'src object'
        };

        c = Ext.clone(o);

        o.src.should.equal('src object');
        c.src.should.equal('src object');

        o.src = 'changed';

        o.src.should.equal('changed');
        c.src.should.equal('src object');
関数型のクローンを生成し、値を変更してオリジナルに影響がないこと


        var o, c;

        o = function() {
            return 'original';
        };

        c = Ext.clone(o);

        o().should.equal('original');
        c().should.equal('original');

        o = function() {
            return 'replace';
        }

        o().should.equal('replace');
        c().should.equal('original');
日付型のクローンを生成し、値を変更してオリジナルに影響がないこと


        var dt, dt2;

        dt = new Date();
        dt2 = Ext.clone(dt);

        dt.getMilliseconds().should.equal(dt2.getMilliseconds());
        dt2.setMilliseconds(50000);
        dt.getMilliseconds().should.not.equal(dt2.getMilliseconds());
配列のクローンを生成し、値を変更してオリジナルに影響がないこと


        var o, c;

        o = function() {
            return 'original';
        };

        c = Ext.clone(o);

        o().should.equal('original');
        c().should.equal('original');

        o = function() {
            return 'replace';
        }

        o().should.equal('replace');
        c().should.equal('original');
オブジェクト内に配列を持つオブジェクトのクローンを生成し、値を変更してオリジナルに影響がないこと


        var o, c;

        o = {
            cn: [{
                tag: 'a'
            },{
                tag: 'span'
            }]
        };
        o.obj = o.cn;

        c = Ext.clone(o);

        o.cn[0].tag.should.equal('a');
        o.cn[1].tag.should.equal('span');

        c.cn[0].tag.should.equal('a');
        c.cn[1].tag.should.equal('span');

        o.obj[0].tag.should.equal('a');
        o.obj[1].tag.should.equal('span');

        c.obj[0].tag.should.equal('a');
        c.obj[1].tag.should.equal('span');

        c.cn.push({tag: 'div'});

        o.cn.length.should.equal(2);
        c.cn.length.should.equal(3);

        o.obj.length.should.equal(2);
        c.obj.length.should.equal(2);
引数に指定する値が、undefined,null,空文字列の場合、引数に指定されたものと同一のものを返却する


        require('assert').equal(Ext.clone(), undefined);
        require('assert').equal(Ext.clone(null), null);
        Ext.clone('').should.equal('');

Ext.Array.each

引数オブジェクトを正常に繰り返すこと


        var sum = function() {
            var sum = 0;

            Ext.Array.each(arguments, function(value) {
                sum += value;
            });

            return sum;
        };

        sum(1, 2, 3).should.equal(6);
配列を正常に繰り返すこと


        var countries = ['Vietnam', 'Singapore', 'United States', 'Russia', 'James'],
            x = 0,
            ret;

        ret = Ext.Array.each(['Vietnam', 'Singapore', 'United States', 'Russia'], function(name, index, countriesItSelf) {
            if(name === 'Singapore' && index === 1 && x === index) {
                return false;
            }
            x++;
        });

        ret.should.equal(1);
配列を逆回転で正常に繰り返すこと


        var countries = ['Vietnam', 'Singapore', 'United States', 'Russia', 'James'],
            x = 0,
            ret;

        ret = Ext.Array.each(countries, function(name, index, countriesItSelf) {
            if(name === 'Singapore' && index === 1 && x === 3) {
                return false;
            }
            x++;
        }, this, true);

        ret.should.equal(1);

Ext.Array.every

        var isBigEnough, passed, x;

        isBigEnough = function(element, index, array) {
            x++;
            return (element >= 10);
        }
        x = 0;
        passed = Ext.Array.every([12, 5, 8, 130, 44], isBigEnough);
        passed.should.equal(false);
        x.should.equal(2);

        x = 0;
        passed = Ext.Array.every([12, 54, 18, 130, 44], isBigEnough);
        passed.should.equal(true);
        x.should.equal(5);
