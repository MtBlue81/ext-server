Ext.apply

空のオブジェクトに値が適用されること


        var src = {}, ret;

        ret = Ext.apply(src, {
            foo: 'bar'
        });

        src.should.equal(ret);
        src.foo.should.equal('bar');
デフォルト値が適用されること


        var src = {}, config, defaults, ret;

        config = {
            foo: 'bar'
        };

        defaults = {
            foo     : 'default value',
            hoge    : 'piyo'
        };

        ret = Ext.apply(src, config, defaults);

        src.should.equal(ret);
        src.foo.should.equal('bar');
        src.hoge.should.equal('piyo');
